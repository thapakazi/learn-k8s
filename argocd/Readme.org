** setup cluster
#+begin_src bash
kind create cluster --name argocd-local --config cluster.yaml
#+end_src

** deploy argocd
#+begin_src bash
  kubectl create namespace argocd
  kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml
#+end_src

** retrieve password
#+begin_src bash
  kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath="{.data.password}" | base64 -d; echo
#+end_src

** port fwd
#+begin_src bash
kubectl port-forward svc/argocd-server -n argocd 6001:443
#+end_src
